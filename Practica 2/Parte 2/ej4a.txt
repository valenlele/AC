PA EQU 30H
PB EQU 31H
CA EQU 32H
CB EQU 33H

ORG 1000H
CHAR DB 'A'

ORG 3000H
INI_PIO: MOV AL, 1
OUT CA, AL
MOV AL, 0
OUT CB, AL
RET

POLL: IN AL, PA
AND AL, 1
JNZ POLL
RET

FLANCO_ASCENDENTE: IN AL, PA
AND AL, 11111101B
OUT PA, AL
OR AL, 00000010B
OUT PA, AL
RET

IMPRIMIR_CARACTER: PUSH AX
CALL POLL
POP AX
OUT PB, AL
CALL FLANCO_ASCENDENTE
RET

ORG 2000H
CALL INI_PIO
MOV AL, CHAR
CALL IMPRIMIR_CARACTER

INT 0
END
