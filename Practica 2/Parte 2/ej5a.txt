DATO EQU 40H
ESTADO EQU 41H

ORG 1000H
STR DB "INGENIERIA E INFORMATICA"
FIN DB ?

ORG 3000H
POLL: IN AL, ESTADO
AND AL, 1
JNZ POLL
RET

IMPRIMIR_CARACTER_HAND: CALL POLL
OUT DATO, AL
RET

ORG 2000H
MOV BX, OFFSET STR
MOV CL, OFFSET FIN - OFFSET STR

LOOP: MOV AL, [BX]
CALL IMPRIMIR_CARACTER_HAND
INC BX
DEC CL
JNZ LOOP

INT 0
END
