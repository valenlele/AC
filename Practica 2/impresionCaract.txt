PA EQU 30H
PB EQU 31H
CA EQU 32H
CB EQU 33H

ORG 1000H
CHAR DB 'A'

ORG 3000H
CONF_PIO:
MOV AL, 01H
OUT CA, AL
MOV AL, 00H
OUT CB, AL
RET

ORG 3100H
STROBE0: IN AL, PA
AND AL, 11111101B
OUT PA, AL
RET

ORG 3200H
STROBE1: IN AL, PA
OR AL, 00000010B
OUT PA, AL
RET

ORG 3300H
POLL: IN AL, PA
AND AL, 1
JNZ POLL
RET

ORG 2000H
CALL CONF_PIO

CALL POLL

MOV AL, CHAR
OUT PB, AL

CALL STROBE0
CALL STROBE1
INT 0
END
