CONT EQU 10H
COMP EQU 11H
PIC EQU 20H
EOI EQU 20H
IMR EQU 21H
INT0 EQU 24H
INT1 EQU 25H

ID_F10 EQU 10
ID_TIMER EQU 20

ORG 40
IP_F10 DW RUT_F10

ORG 80
IP_TIMER DW RUT_TIMER

ORG 1000H
VALOR DB ?
ESPACIO DB ' '
CUENTA DB ?
FLAG DB 0

ORG 2000H
CLI

MOV AL, 0FCH
OUT IMR, AL

MOV AL, ID_F10
OUT INT0, AL

MOV AL, ID_TIMER
OUT INT1, AL

MOV BX, OFFSET VALOR
INT 6

SUB BYTE PTR [BX], 30H
MOV DL, [BX]
ADD BYTE PTR [BX], 30H

MOV AL, 1
OUT COMP, AL

STI

LOOP: CMP DL, 0
JNZ LOOP

INT 0

ORG 3000H
RUT_F10: MOV AL, 0
OUT CONT, AL

MOV AL, EOI
OUT PIC, AL
IRET

ORG 3100H
RUT_TIMER: MOV AL, 1
INT 7
MOV CX, BX
MOV BX, OFFSET ESPACIO
INT 7
MOV BX, CX
DEC BYTE PTR [BX]
DEC DL

CMP DL, 0FFH
JZ FIN

MOV AL, 0
OUT CONT, AL

FIN: MOV AL, EOI
OUT PIC, AL
IRET

END
