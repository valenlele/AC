CONT EQU 10H
COMP EQU 11H
PIC EQU 20H
EOI EQU 20H
IMR EQU 21H
INT0 EQU 24H
INT1 EQU 25H

ID_F10 EQU 10
ID_TIMER EQU 20

ORG 40
IP_F10 DW RUT_F10

ORG 80
IP_TIMER DW RUT_TIMER

ORG 2000H

MOV AL, ID_F10
OUT INT0, AL

MOV AL, ID_TIMER
OUT INT1, AL

MOV AL, 0FDH
OUT IMR, AL

MOV AL, 1
OUT COMP, AL

MOV AL, 0
OUT CONT, AL

LOOP: JMP LOOP

ORG 3000H
RUT_F10:
IRET

ORG 3100H
RUT_TIMER: MOV AL, 1
INT 7

MOV AL, EOI
OUT PIC, AL
IRET

END
