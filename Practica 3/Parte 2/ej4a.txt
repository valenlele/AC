;Escribir un programa que muestre en pantalla “Vamos las interrupciones!” 
;cada vez que se presiona la tecla F10. El programa nunca termina
PIC EQU 20H
EOI EQU 20H
IMR EQU 21H
INT0 EQU 24H
N_F10 EQU 10

ORG 40
IP_F10 DW RUT_F10

ORG 1000H
STR DB "Vamos las interrupciones!"
FIN DB ?

ORG 2000H
CLI
MOV AL, 0FEH
OUT IMR, AL
MOV AL, N_F10
OUT INT0, AL
STI
LOOP: JMP LOOP

ORG 3000H
RUT_F10: PUSH AX
PUSH BX
MOV AL, OFFSET FIN - OFFSET STR
MOV BX, OFFSET STR
INT 7
MOV AL, EOI
OUT PIC, AL
POP BX
POP AX
IRET

END
