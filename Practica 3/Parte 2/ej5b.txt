PIC EQU 20H
EOI EQU 20H
IMR EQU 21H
INT1 EQU 25H

CONT EQU 10H
COMP EQU 11H

ID_TIMER EQU 20

ORG 80
IP_TIMER DW RUT_TIMER

ORG 1000H
STR DB "Vamos las interrupciones!"
FIN DB ?
CANT DB 0

ORG 2000H
CLI
MOV AL, 0FDH
OUT IMR, AL

MOV AL, ID_TIMER
OUT INT1, AL

MOV AL, 0
OUT CONT, AL

MOV AL, 2
OUT COMP, AL

STI

LOOP: CMP CANT, 10
JNZ LOOP

INT 0

ORG 3000H
RUT_TIMER: PUSH AX
PUSH BX

MOV AL, OFFSET FIN - OFFSET STR
MOV BX, OFFSET STR
INT 7

ADD CANT, 2

MOV AL, 0
OUT CONT, AL

MOV AL, EOI
OUT PIC, AL

POP BX
POP AX
IRET

END
