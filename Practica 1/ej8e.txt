ORG 3000H
CONTAR_CAR: PUSH AX
PUSH BX

MOV BX, SP
MOV CH, 0

ADD BX, 8
MOV AL, [BX]

SUB BX, 2  
MOV BX, [BX]

LOOP: MOV CL, BYTE PTR [BX]
CMP CL, 00h
JZ FIN

CMP CL, AL
JNZ LOOP2

INC CH

LOOP2: INC BX
JMP LOOP

FIN: POP BX
POP AX
RET
