CADENA DB "oRGAnizacion", 00h
CANT DB ?

org 3000h
es_mayus: cmp ah, dh
js false

cmp dl, ah
js false

mov dh, 0FFh ;devuelve 0FFh si es mayus
jmp fin

false: mov dh, 00h

fin: ret

ES_VOCAL: mov dh, 'A'
mov dl, 'Z'
call es_mayus

cmp dh, 0FFh ;verifica si es mayus
jnz validar ;si no da 0, es minus. no hace falta el pasaje

add ah, 32 ;si no salto a validar es mayus, se hace el pasaje

validar: cmp ah, 'a'
jz true2

cmp ah, 'e'
jz true2

cmp ah, 'i'
jz true2

cmp ah, 'o'
jz true2

cmp ah, 'u'
jz true2

mov ah, 00h
jmp fin2

true2: mov ah, 0FFh

fin2: ret

CONTAR_VOC: MOV CH, 0
LOOP: MOV AH, [BX]
CMP AH, 00h
JZ FIN3

PUSH DX
CALL ES_VOCAL
POP DX
CMP AH, 0FFh
JNZ LOOP2
INC CH

LOOP2: INC BX
JMP LOOP

FIN3: RET

org 2000h
mov BX, offset cadena
PUSH AX
PUSH CX
CALL CONTAR_VOC
POP AX
POP CX
MOV CANT, CH
hlt
