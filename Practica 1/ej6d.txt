ORG 1000H
NUM1 DB 3
NUM2 DB 2
RES DB 0

ORG 3000H
MUL: MOV BX, SP
ADD BX, 3
CMP BYTE PTR [BX], 0
JZ FIN
SUB BX, 1
CMP BYTE PTR [BX], 0
JZ FIN

ADD BX, 1
MOV DH, [BX]
SUB BX, 1

LOOP: ADD DL, DH
DEC BYTE PTR [BX]
JNZ LOOP

FIN:
RET

ORG 2000H
MOV AL, NUM1
MOV AH, NUM2
PUSH AX
MOV DL, RES
CALL MUL
MOV RES, DL
POP AX
HLT
END
