ORG 3000H
MUL: MOV BX, SP
ADD BX, 4
CMP BYTE PTR [BX], 0
JZ FIN
SUB BX, 2
CMP BYTE PTR [BX], 0
JZ FIN

ADD BX, 2
MOV DH, [BX]
SUB BX, 2

LOOP: ADD DL, DH
DEC BYTE PTR [BX]
JNZ LOOP

FIN:
RET
