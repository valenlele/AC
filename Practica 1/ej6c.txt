ORG 1000H
NUM1 DB 3
NUM2 DB 2
RES DB 0

ORG 3000H
MUL: MOV BX, AX
CMP BYTE PTR [BX], 0
JZ FIN
MOV BX, CX
CMP BYTE PTR [BX], 0
JZ FIN

MOV DH, [BX]

LOOP: MOV BX, AX
ADD DL, [BX]
MOV BX, CX
DEC DH
JNZ LOOP
FIN:
RET

ORG 2000H
MOV AX, OFFSET NUM1
MOV CX, OFFSET NUM2
MOV DL, RES
CALL MUL
MOV RES, DL
HLT
END
