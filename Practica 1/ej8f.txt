ORG 1000H
ORIGINAL DB "A"
REEMPLAZO DB "O"
CADENA DB "HAlO", 00h

ORG 3000H
REEMPLAZAR_CAR: PUSH AX
PUSH BX

MOV BX, SP
ADD BX, 5
MOV AH, [BX]
DEC BX
MOV AL, [BX]
SUB BX, 2
MOV BX, [BX]

LOOP: CMP BYTE PTR [BX], 00h
JZ FIN
CMP AH, [BX]
JNZ SEGUIR

MOV [BX], AL

SEGUIR: INC BX
JMP LOOP

FIN: POP BX
POP AX
RET

ORG 2000H
MOV AH, ORIGINAL
MOV AL, REEMPLAZO
PUSH AX
MOV AX, OFFSET CADENA
PUSH AX
CALL REEMPLAZAR_CAR
HLT
END
