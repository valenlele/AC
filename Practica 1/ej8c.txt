ORG 3000H
ES_VOCAL: MOV AH, 'A'
CMP AH, AL
JZ SI
ADD AH, 32
CMP AH, AL
JZ SI

MOV AH, 'E'
CMP AH, AL
JZ SI
ADD AH, 32
CMP AH, AL
JZ SI

MOV AH, 'I'
CMP AH, AL
JZ SI
ADD AH, 32
CMP AH, AL
JZ SI

MOV AH, 'O'
CMP AH, AL
JZ SI
ADD AH, 32
CMP AH, AL
JZ SI

MOV AH, 'U'
CMP AH, AL
JZ SI
ADD AH, 32
CMP AH, AL
JZ SI
JMP NO

SI: MOV AL, 0FFh
JMP FIN

NO: MOV AL, 00h

FIN: RET

ORG 2000H
MOV AL, CAR

MOV RES, AL
HLT
END
